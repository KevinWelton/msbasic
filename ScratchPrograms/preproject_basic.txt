REM This is the program from the "Hacking Microsoft BASIC" video by Ben Eater, prior to custom BASIC instructions being added.

1000 REM LCDINIT
1010 POKE 24578, 255
1020 POKE 24576, 3
1030 POKE 24576, 67
1040 POKE 24576, 3
1050 POKE 24576, 67
1060 POKE 24576, 3
1070 POKE 24576, 67
1080 POKE 24576, 3
1090 POKE 24576, 2
1100 POKE 24576, 66
1110 POKE 24576, 2
1120 RETURN

1200 REM LCDCMD
1210 POKE 24576, INT(CMD/16)
1220 POKE 24576, INT(CMD/16) OR 64
1230 POKE 24576, INT(CMD/16)
1240 POKE 24576, CMD AND 15
1250 POKE 24576, (CMD AND 15) OR 64
1260 POKE 24576, CMD AND 15
1270 RETURN

1300 REM LCDPRINT
1310 POKE 24576, INT(ASC(CHAR$)/16) OR 16
1320 POKE 24576, INT(ASC(CHAR$)/16) OR 16 OR 64
1330 POKE 24576, INT(ASC(CHAR$)/16) OR 16
1340 POKE 24576, (ASC(CHAR$) AND 15) OR 16
1350 POKE 24576, (ASC(CHAR$) AND 15) OR 16 OR 64
1360 POKE 24576, (ASC(CHAR$) AND 15) OR 16
1370 RETURN

10 GOSUB 1000
20 CMD = 40
30 GOSUB 1200
40 CMD = 14
50 GOSUB 1200
60 CMD = 6
70 GOSUB 1200
80 CMD = 1
90 GOSUB 1200
110 S$="Hello, world!"
120 FOR I = 1 TO LEN(S$)
130 CHAR$ = MID$(S$, I, 1)
140 GOSUB 1300
150 NEXT I
160 END
